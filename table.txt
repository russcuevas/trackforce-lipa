
CREATE TABLE incidents (
    id INT AUTO_INCREMENT PRIMARY KEY,
    report_number VARCHAR(20) UNIQUE NOT NULL, -- Format: TFL-YYYY-XXXX
    incident_type VARCHAR(255) NOT NULL,

    -- Location & Environment (From Chapter 4 Findings)
    location_name VARCHAR(255) NOT NULL,
    latitude DECIMAL(10, 8) NOT NULL,
    longitude DECIMAL(11, 8) NOT NULL,
    road_condition VARCHAR(100) NULL, -- e.g., Wet, Dry, Under Construction
    weather_condition VARCHAR(100) NULL, -- e.g., Clear, Raining, Foggy
    
    -- Reporter (Citizen/First Responder)
    reporter_name VARCHAR(100) NULL,
    reporter_contact VARCHAR(20) NULL,
    reporter_email VARCHAR(20) NULL,
    reporter_address TEXT NULL,
    is_first_responder BOOLEAN DEFAULT FALSE,
    
    -- Status & Analytics
    status VARCHAR(100),
    time_reported TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    time_documented DATETIME NULL,
    time_completed DATETIME NULL,
    otp VARCHAR(20) NOT NULL,
    is_verified
    -- Assignment
    assigned_investigator_id INT NULL,
    FOREIGN KEY (assigned_investigator_id) REFERENCES users(user_id) ON DELETE SET NULL
);

CREATE TABLE involved_parties (
    id INT AUTO_INCREMENT PRIMARY KEY,
    incident_id INT NOT NULL,
    full_name VARCHAR(100) NULL,
    age INT NULL,
    sex VARCHAR(100) NULL,
    role VARCHAR(100) NULL,
    license_number VARCHAR(50) NULL,
    injury_severity VARCHAR(100) NULL,
    statement TEXT NULL,
    FOREIGN KEY (incident_id) REFERENCES incidents(incident_id) ON DELETE CASCADE
);

CREATE TABLE vehicles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    incident_id INT NOT NULL,
    plate_number VARCHAR(20) NULL,
    vehicle_type VARCHAR(30) NULL,
    color VARCHAR(30) NULL,
    FOREIGN KEY (incident_id) REFERENCES incidents(incident_id) ON DELETE CASCADE
);


CREATE TABLE incident_evidence (
    evidence_id INT AUTO_INCREMENT PRIMARY KEY,
    incident_id INT NOT NULL,
    file_path VARCHAR(255) NOT NULL, -- Storage path for photos/videos
    file_type VARCHAR(50) NULL,
    uploaded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (incident_id) REFERENCES incidents(incident_id) ON DELETE CASCADE
);

CREATE TABLE audit_logs (
    log_id INT AUTO_INCREMENT PRIMARY KEY,
    incident_id INT NOT NULL,
    user_id INT NOT NULL,
    action_performed TEXT NOT NULL, -- e.g., "Updated status to Solved"
    timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (incident_id) REFERENCES incidents(incident_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(user_id)
);
